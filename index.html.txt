<script>

const W=1.22
const L=2.44

document.querySelectorAll("input").forEach(i=>{
i.addEventListener("input",calc)
})

function num(id){

return parseFloat(
document.getElementById(id).value.replace(/,/g,'')
)||0

}

function fmt(x){

return Number(x).toLocaleString("en-US",{maximumFractionDigits:2})

}

function calc(){

let t=num("thickness")

let bundle=num("bundle")

let pcsb=num("pcsBundle")

let usd=num("usd")

let exwpcs=num("exwpcs")

let exwcbm=num("exwcbm")

let logistics=num("logistics")

let cfr=num("cfr")

let pcs=bundle*pcsb

let cbmBuyPc=(t/1000)*W*L

let cbmSellPc=((t-0.5)/1000)*W*L

let cbmBuy=pcs*cbmBuyPc

let cbmSell=pcs*cbmSellPc

let cbmDiff=cbmSell-cbmBuy

let exwTotal=0

if(exwpcs>0){

exwTotal=pcs*exwpcs

}

else if(exwcbm>0){

exwTotal=cbmBuy*exwcbm

}

let totalCfr=exwTotal+logistics

let selling=cbmSell*cfr*usd

let profit=Math.round(selling-totalCfr)

document.getElementById("pcs").innerText=fmt(pcs)

document.getElementById("buy").innerText=fmt(cbmBuy)

document.getElementById("sell").innerText=fmt(cbmSell)

document.getElementById("diff").innerText=fmt(cbmDiff)

let p=document.getElementById("profit")

p.innerText=fmt(profit)

p.className=profit>=0?"blue":"red"

}
